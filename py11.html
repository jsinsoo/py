01. 웹페이지 제목 가져오기

import requests
from bs4 import BeautifulSoup

url = "https://example.com"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")

print("웹페이지 제목:", soup.title.text)

상세 설명

가장 기초적인 크롤링 예제입니다. requests.get()으로 웹페이지 HTML을 가져오고, BeautifulSoup으로 HTML을 분석한 뒤 title 태그의 텍스트를 출력합니다.

02. 뉴스 제목 크롤링

import requests
from bs4 import BeautifulSoup

url = "https://news.ycombinator.com/"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")

titles = soup.find_all("span", class_="titleline")

for i, title in enumerate(titles[:10], 1):
    print(i, title.text)

상세 설명

뉴스나 게시글 제목처럼 반복되는 HTML 요소를 수집하는 예제입니다. find_all()은 조건에 맞는 모든 태그를 리스트 형태로 가져옵니다.

03. 모든 링크 주소 추출

import requests
from bs4 import BeautifulSoup

url = "https://example.com"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")

links = soup.find_all("a")

for link in links:
    print(link.get("href"))

상세 설명

a 태그는 보통 링크를 의미합니다. href 속성에 실제 이동 주소가 들어 있습니다. link.get("href")는 href 속성이 없을 때도 오류를 줄일 수 있는 안전한 방식입니다.

04. 쇼핑몰 상품 가격 수집

import requests
from bs4 import BeautifulSoup

url = "https://books.toscrape.com/"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")

prices = soup.find_all("p", class_="price_color")

for price in prices[:10]:
    print(price.text)

상세 설명

연습용 사이트인 books.toscrape.com에서 책 가격 정보를 가져오는 예제입니다. 상품 가격, 책 제목, 평점 등을 함께 수집하면 가격 비교 데이터셋을 만들 수 있습니다.

05. 이미지 주소 수집

from bs4 import BeautifulSoup

soup = BeautifulSoup(res.text, "html.parser")

06. 크롤링 결과 CSV 저장

import requests
from bs4 import BeautifulSoup
import csv

url = "https://news.ycombinator.com/"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")
titles = soup.find_all("span", class_="titleline")

with open("news.csv", "w", newline="", encoding="utf-8-sig") as f:
    writer = csv.writer(f)
    writer.writerow(["번호", "제목"])

    for i, title in enumerate(titles[:10], 1):
        writer.writerow([i, title.text])

상세 설명

수집한 데이터를 파일로 저장하는 예제입니다. CSV로 저장하면 엑셀, 구글 스프레드시트, Pandas에서 쉽게 열어 분석할 수 있습니다. 한글 깨짐 방지를 위해 utf-8-sig를 사용했습니다.

07. 날씨 정보 크롤링 구조

import requests
from bs4 import BeautifulSoup

url = "https://example.com/weather"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")

temp = soup.find("span", class_="temp")

if temp:
    print("현재 기온:", temp.text)
else:
    print("기온 정보를 찾을 수 없습니다.")

상세 설명

날씨 사이트의 기온 정보를 가져오는 구조 예제입니다. 실제 사이트마다 태그와 class 이름이 다르므로 개발자 도구로 HTML 구조를 확인해야 합니다. 조건문을 넣어 태그를 찾지 못했을 때 오류를 방지합니다.

08. Selenium 로그인 자동화 구조

from selenium import webdriver
from selenium.webdriver.common.by import By
import time

driver = webdriver.Chrome()

driver.get("https://example.com/login")

driver.find_element(By.NAME, "username").send_keys("myid")
driver.find_element(By.NAME, "password").send_keys("mypassword")

time.sleep(2)

driver.quit()

상세 설명

Selenium은 실제 브라우저를 자동으로 조작합니다. 로그인, 버튼 클릭, 검색 입력, 무한 스크롤처럼 JavaScript가 필요한 작업에 사용됩니다. 실제 로그인 자동화는 사이트 정책을 확인한 뒤 허가된 환경에서만 사용해야 합니다.

09. 무한 스크롤 페이지 크롤링

from selenium import webdriver
import time

driver = webdriver.Chrome()
driver.get("https://example.com")

for _ in range(5):
    driver.execute_script(
        "window.scrollTo(0, document.body.scrollHeight);"
    )
    time.sleep(2)

print(driver.page_source)

driver.quit()

상세 설명

무한 스크롤 페이지는 아래로 내려야 새 내용이 로딩됩니다. Selenium의 execute_script()로 브라우저 스크롤을 내리고, page_source로 로딩된 HTML을 확인할 수 있습니다.

10. 크롤링 → Pandas 분석

import requests
from bs4 import BeautifulSoup
import pandas as pd

url = "https://books.toscrape.com/"
res = requests.get(url)

soup = BeautifulSoup(res.text, "html.parser")

titles = soup.find_all("h3")

data = []

for t in titles:
    data.append(t.text.strip())

df = pd.DataFrame(data, columns=["책 제목"])

print(df)

상세 설명

크롤링한 데이터를 Pandas DataFrame으로 변환하는 예제입니다. 이후 CSV 저장, 빈도 분석, 시각화, 머신러닝 데이터셋 생성으로 확장할 수 있습니다.
